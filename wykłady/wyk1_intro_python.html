#      x<0.1  => 1
# 0.1<=x<0.4  => 2
# 0.4<=x<0.7  => 3
# 0.7<=x      => 4
def wybor(x):
    if x<0.1:
        i=1
    elif x<0.4:
        i=2
    elif x<0.7:
        i=3
    else:
        i=4
    return i

# deklarowanie wasnych funkcji
# utworzymy funkcj bisek, ktra dla dowolnej innej funkcji f(x),
# majcej przeciwne znaki na kocach przedziau (a,b) -
# f(a)*f(b)<0,zwrci o poow krtszy przedzia z  przeciwnymi 
# znakami f na jego kocach
# 
def bisek(f,a,b):
    if f(a)*f(b)<0:
        c=(a+b)/2
        if f(a)*f(c)<0:
            return a,c
        else:
            return c,b
# wielomian 3. st. z miejscami zerowymi w x=1, 2 i 5
wiel3=lambda x:(x+1)*(x-2)*(x-5)

#sprawdzimy dziaanie funkcji bisek w przedziale (0, 3) 
# zawierajcym pierwiastek x=1 
a=0.0
b=3
a,b=bisek(wiel3,a,b)
print("%16.12f  %16.12f  %16.12f" % (a,b,b-a))
while b-a>0.001:
    a,b=bisek(wiel3,a,b)
# formatowanie wydruku 
    print("%16.12f  %16.12f  %16.12f" % (a,b,b-a))
